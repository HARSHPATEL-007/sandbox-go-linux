# Stage 1: Build nsjail and Go binary
FROM ubuntu:22.04 AS builder

# Install nsjail dependencies & Go
RUN apt-get update && apt-get install -y \
    autoconf bison flex gcc g++ git libprotobuf-dev libnl-route-3-dev libtool \
    make pkg-config protobuf-compiler golang-go

WORKDIR /build

# Strict requirement: fetch and build nsjail from git tag 3.4
RUN git clone -b 3.4 https://github.com/google/nsjail.git && \
    cd nsjail && \
    make && \
    mv nsjail /usr/bin/nsjail

# Build the Go application
COPY go.mod go.sum ./
RUN go mod download
COPY . .
RUN CGO_ENABLED=0 GOOS=linux go build -o goboxd ./cmd/goboxd/main.go

# Stage 2: Final runtime image
FROM ubuntu:22.04

# Install supported toolchains (Python, C++, Java, Node, Bash is built-in)
RUN apt-get update && apt-get install -y \
    python3 python3-pip \
    g++ \
    openjdk-17-jdk \
    nodejs \
    iverilog \
    && rm -rf /var/lib/apt/lists/*

# Copy nsjail from builder
COPY --from=builder /usr/bin/nsjail /usr/bin/nsjail

# Copy application and config
WORKDIR /app
COPY --from=builder /build/goboxd .
COPY config/ config/

EXPOSE 8080
CMD ["./goboxd"]
